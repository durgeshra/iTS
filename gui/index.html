#include "iostream" using namespace std; int main() { int T, N, M, A1, A2, A3, Ans; cin >> T; while(T--) { cin >> N >> M; cin >> A1 >> A2 >> A3; int A[N]; A[0] = A1 % M; A[1] = A2 % M; A[2] = A3 % M; Ans = A1 + A2 + A3; for(int i = 3; i < N; i = i + 1) { A[i] = (A[i-1] + A[i-2] + A[i-3]) % M; Ans = (Ans + A[i]) % M; } cout << Ans << endl; } return 0; }
